int sub_411177(int a1, int a2)
{
    int v2; // edx

    int v4; // ebx

    v2 = 0;
    v3 = 0;
    if ( a2 > 0 )
    {
        v4 = a2 - 1;

            v2 |= ((a1 >> v3++) & 1) << v4--;
        while ( v3 < a2 );
    }
    return v2 + 1;
}

void v19(int *key)
